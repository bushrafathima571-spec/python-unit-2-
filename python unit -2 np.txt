# Program to calculate factorial using recursion

def factorial(n):
    if n == 0 or n == 1:
        return 1
    return n * factorial(n - 1)

print(f"Factorial of {num}:", factorial(num))

Output:

Factorial of 5: 120

9. Write a Python program to define a function that accepts two numbers and returns their greatest common divisor (GCD).

def find_gcd(a, b):
    while b != 0:
        a, b = b, a % b
    return a

gcd = find_gcd(num1, num2)

print(f"The GCD of {num1} and {num2} is {gcd}")

output:

Enter second number: 4
The GCD of 2 and 4 is 2

11.Write a Python program demonstrating single inheritance using a base class Person and a derived class Student.

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def display_person(self):
        print(f"Name: {self.name}")
        print(f"Age: {self.age}")

class Student(Person):

    def __init__(self, name, age, course):

        super().__init__(name, age)
        self.course = course

    def display_student(self):
        self.display_person()
        print(f"Course: {self.course}")


name = input("Enter name: ")
age = int(input("Enter age: "))
course = input("Enter course: ")

student = Student(name, age, course)


print("\n--- Student Details ---")
student.display_student()

output:
Enter name: sanju
Enter age: 17
Enter course: ABC

--- Student Details ---
Name: sanju
Age: 17
Course: ABC

27.To check whether a string is palindrome using recursion.

def palindrome(text):

    if len(text) <= 1:
        return True

    if text[0] == text[-1]:
        return palindrome(text[1:-1])
    else:
        return False

word = "madam"

if palindrome(word):
    print(word, "is Palindrome")
else:
    print(word, "is Not Palindrome")
Output:
madam is Palindrome
